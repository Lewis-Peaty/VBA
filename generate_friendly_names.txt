'PURPOSE: Remove table name and brackets to create friendly field names.
'         Only works on correctly formatted full-field names.
'	  E.g. [Failures].[Fault Cause] becomes Fault Cause
Sub generate_friendly_names()
    Application.Calculation = xlCalculationManual
    Application.ScreenUpdating = False

    Dim full As Range  'full names
    Set full = Range("A2", Cells(2, 1).End(xlDown))
    Dim fullStopPos As Integer
    Dim intermediateString As String
    For Each c In full.Cells
        fullStopPos = InStr(c.Value, ".")
        intermediateString = Right(c.Value, Len(c.Value) - fullStopPos - 1)
        c.Offset(0, 1).Value = Left(intermediateString, Len(intermediateString) - 1)
    Next c

    Application.Calculation = xlCalculationAutomatic
    Application.ScreenUpdating = True
End Sub
