Sub concatStrings()

'Grab values from col 2 which ended up in col 1 due to newline characters, move to col 2
 For Each c In Range("A1:A3000")
    If Not (c.Value Like "INCD*") Then
        c.Offset(0, 1) = c.Value
        c.ClearContents
    End If
 Next c

'Concat values in col 2 with no corresponding col 1 value and concat them with preceeding row
 For i = 3000 To 3 Step -1
    If Not (Range("A" & i).Value Like "INCD*") Then
        Range("B" & (i - 1)) = Range("B" & (i - 1)) & " " & Range("B" & i) 'Concat
        Range("B" & i).ClearContents
    End If
 Next i

End Sub
